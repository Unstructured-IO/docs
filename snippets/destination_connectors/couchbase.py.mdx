import os

from unstructured.ingest.connector.couchbase import (
    CouchbaseAccessConfig,
    CouchbaseWriteConfig,
    SimpleCouchbaseConfig,
)
from unstructured.ingest.connector.local import SimpleLocalConfig
from unstructured.ingest.interfaces import (
    ChunkingConfig,
    EmbeddingConfig,
    PartitionConfig,
    ProcessorConfig,
    ReadConfig,
)
from unstructured.ingest.runner import LocalRunner
from unstructured.ingest.runner.writers.base_writer import Writer
from unstructured.ingest.runner.writers.couchbase import (
    CouchbaseWriter,
)


def get_writer() -> Writer:
    return CouchbaseWriter(
        connector_config=SimpleCouchbaseConfig(
            access_config=CouchbaseAccessConfig(
                connection_string=os.getenv("COUCHBASE_CONNECTION_STRING"),
                username=os.getenv("COUCHBASE_USERNAME"),
                password=os.getenv("COUCHBASE_PASSWORD"),
            ),
            bucket=os.getenv("COUCHBASE_BUCKET"),
            scope=os.getenv("COUCHBASE_SCOPE"),
            collection=os.getenv("COUCHBASE_COLLECTION"),
        ),
        write_config=CouchbaseWriteConfig(),
    )


if __name__ == "__main__":
    writer = get_writer()
    runner = LocalRunner(
        processor_config=ProcessorConfig(
            verbose=True,
            output_dir="local-output-to-couchbase",
            num_processes=2,
        ),
        connector_config=SimpleLocalConfig(
            input_path="example-docs/book-war-and-peace-1225p.txt",
        ),
        read_config=ReadConfig(),
        partition_config=PartitionConfig(),
        chunking_config=ChunkingConfig(chunk_elements=True),
        embedding_config=EmbeddingConfig(
            provider="langchain-huggingface",
        ),
        writer=writer,
        writer_kwargs={},
    )
    runner.run()
