import os
from unstructured.ingest.interfaces import WriteConfig
from unstructured.ingest.connector.local import SimpleLocalConfig
from unstructured.ingest.connector.sql import (
    SqlAccessConfig,
    # SqlWriteConfig,
    SimpleSqlConfig,
)
from unstructured.ingest.interfaces import (
    ChunkingConfig,
    EmbeddingConfig,
    PartitionConfig,
    ProcessorConfig,
    ReadConfig,
)
from unstructured.ingest.runner import LocalRunner
from unstructured.ingest.runner.writers.base_writer import Writer
from unstructured.ingest.runner.writers.sql import (
    SqlWriter,
)


def get_writer() -> Writer:
    return SqlWriter(
        connector_config=SimpleSqlConfig(
            access_config=SqlAccessConfig(username="unstructured", password="test"),
            db_type="postgresql",
            host="localhost",
            database="elements",
            port=5433,
        ),
        write_config=WriteConfig(),
    )


if __name__ == "__main__":
    writer = get_writer()
    runner = LocalRunner(
        processor_config=ProcessorConfig(
            verbose=True,
            output_dir="local-output-to-sql",
            num_processes=1,
            work_dir="local-work-to-sql",
        ),
        connector_config=SimpleLocalConfig(
            input_path="example-docs/fakememo/",
        ),
        read_config=ReadConfig(),
        partition_config=PartitionConfig(
            # metadata_exclude=["orig_elements"],
        ),
        chunking_config=ChunkingConfig(chunk_elements=True, include_orig_elements=False),
        embedding_config=EmbeddingConfig(
            provider="langchain-huggingface",
            api_key=None,
        ),
        writer=writer,
        writer_kwargs={},
    )
    runner.run()
