    response = client.sources.create_source(
        request=CreateSourceRequest(
            create_source_connector=CreateSourceConnector(
                name="<name>",
                type="teradata",
                config={
                    "host": "<host>",
                    "database": "<database>",
                    "table_name": "<table-name>",
                    "batch_size": <batch-size>,
                    "id_column": "<id-column>",
                    "column_names": [
                        "<column-name>",
                        "<column-name>"
                    ],
                    "username": "<username>",
                    "password": "<password>"
                }
            )
        )
    )

    print(response.source_connector_information)
