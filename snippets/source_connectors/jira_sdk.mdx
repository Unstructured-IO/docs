                    password="<password">,

                    # For personal access token authentication:
                    token="<token>",
        
                    projects=[ 
                        "<project-id>",
                        "<project-id>"
                    ],
                    boards=[ 
                        "<board-id>",
                        "<board-id>"
                    ],
                    issues=[ 
                        "<issue-id>",
                        "<issue-id>"
                    ],

                )
