```bash CLI
#!/usr/bin/env bash

unstructured-ingest \
  gcs \
    --service-account-key $GCS_SERVICE_ACCOUNT_KEY \
    --remote-url $GCS_REMOTE_URL \
    --output-dir $LOCAL_FILE_OUTPUT_DIR \
    --num-processes 2 \
    --recursive \
    --verbose \
    --strategy hi_res \
    --partition-by-api \
    --api-key $UNSTRUCTURED_API_KEY \
    --partition-endpoint $UNSTRUCTURED_API_URL
```
