```bash CLI
#!/usr/bin/env bash

unstructured-ingest \
  google-drive \
    --download-dir google-drive-ingest-download \
    --drive-id $GOOGLE_DRIVE_FOLDER_ID \
    --service-account-key $GCP_SERVICE_ACCOUNT_KEY \
    --partition-by-api \
    --api-key $UNSTRUCTURED_API_KEY \
    --partition-endpoint $UNSTRUCTURED_API_URL \
    --strategy fast \
    --chunking-strategy by_title \
    --embedding-provider langchain-huggingface \
  local \
    --output-dir google-drive-ingest-output
```
